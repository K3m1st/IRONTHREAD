  <script>
  (function () {
    /* ── NAV ── */
    var nav = document.getElementById('nav');
    var burger = document.getElementById('burger');
    var navLinks = document.getElementById('navLinks');
    window.addEventListener('scroll', function () {
      nav.classList.toggle('nav--scrolled', scrollY > 40);
    });
    burger.addEventListener('click', function () {
      burger.classList.toggle('open');
      navLinks.classList.toggle('open');
    });
    var anchors = navLinks.querySelectorAll('a');
    for (var i = 0; i < anchors.length; i++) {
      anchors[i].addEventListener('click', function () {
        burger.classList.remove('open');
        navLinks.classList.remove('open');
      });
    }

    /* ── SMOOTH SCROLL ── */
    var scrollLinks = document.querySelectorAll('a[href^="#"]');
    for (var s = 0; s < scrollLinks.length; s++) {
      scrollLinks[s].addEventListener('click', function (e) {
        e.preventDefault();
        var target = document.querySelector(this.getAttribute('href'));
        if (target) target.scrollIntoView({ behavior: 'smooth' });
      });
    }

    /* ── HERO ANIM STAGGER ── */
    var anims = document.querySelectorAll('.anim');
    for (var a = 0; a < anims.length; a++) {
      anims[a].style.animationDelay = (parseInt(anims[a].dataset.d) * 0.12 + 0.3) + 's';
    }

    /* ── SCROLL REVEAL ── */
    var io = new IntersectionObserver(function (entries) {
      entries.forEach(function (e) {
        if (e.isIntersecting) {
          e.target.classList.add('in');
          io.unobserve(e.target);
        }
      });
    }, { threshold: 0.12, rootMargin: '0px 0px -60px 0px' });
    var revealEls = document.querySelectorAll('.reveal');
    for (var r = 0; r < revealEls.length; r++) {
      revealEls[r].style.transitionDelay = ((r % 4) * 0.09) + 's';
      io.observe(revealEls[r]);
    }

    /* ── HERO CANVAS — TOPOLOGY NETWORK ── */
    var hc = document.getElementById('heroCanvas');
    var hctx = hc.getContext('2d');
    var nodes = [];
    var hW, hH;
    var mouse = { x: -9999, y: -9999 };

    function hResize() {
      var hero = document.getElementById('hero');
      hW = hc.width = hero.offsetWidth;
      hH = hc.height = hero.offsetHeight;
      nodes = [];
      var count = Math.min(60, Math.floor(hW * hH / 18000));
      for (var i = 0; i < count; i++) nodes.push(makeNode());
    }

    function makeNode() {
      return {
        x: Math.random() * hW,
        y: Math.random() * hH,
        vx: (Math.random() - 0.5) * 0.25,
        vy: (Math.random() - 0.5) * 0.25,
        r: Math.random() * 2 + 1,
        pulse: Math.random() * Math.PI * 2
      };
    }

    hResize();
    window.addEventListener('resize', hResize);
    document.getElementById('hero').addEventListener('mousemove', function (e) {
      var rect = hc.getBoundingClientRect();
      mouse.x = e.clientX - rect.left;
      mouse.y = e.clientY - rect.top;
    });
    document.getElementById('hero').addEventListener('mouseleave', function () {
      mouse.x = -9999; mouse.y = -9999;
    });

    function heroFrame(t) {
      hctx.clearRect(0, 0, hW, hH);
      var time = t * 0.001;

      for (var i = 0; i < nodes.length; i++) {
        var n = nodes[i];
        n.x += n.vx;
        n.y += n.vy;
        if (n.x < 0 || n.x > hW) n.vx *= -1;
        if (n.y < 0 || n.y > hH) n.vy *= -1;

        var dx = mouse.x - n.x, dy = mouse.y - n.y;
        var md = Math.sqrt(dx * dx + dy * dy);
        if (md < 200) {
          n.x -= dx * 0.008;
          n.y -= dy * 0.008;
        }

        var pulse = Math.sin(time * 1.5 + n.pulse) * 0.3 + 0.7;
        hctx.beginPath();
        hctx.arc(n.x, n.y, n.r * pulse, 0, Math.PI * 2);
        hctx.fillStyle = 'rgba(0,240,255,' + (0.5 * pulse) + ')';
        hctx.fill();

        // glow
        hctx.beginPath();
        hctx.arc(n.x, n.y, n.r * 3 * pulse, 0, Math.PI * 2);
        hctx.fillStyle = 'rgba(0,240,255,' + (0.06 * pulse) + ')';
        hctx.fill();
      }

      // connections
      for (var a = 0; a < nodes.length; a++) {
        for (var b = a + 1; b < nodes.length; b++) {
          var ddx = nodes[a].x - nodes[b].x;
          var ddy = nodes[a].y - nodes[b].y;
          var dd = ddx * ddx + ddy * ddy;
          if (dd < 22000) {
            var alpha = 0.12 * (1 - dd / 22000);
            hctx.beginPath();
            hctx.moveTo(nodes[a].x, nodes[a].y);
            hctx.lineTo(nodes[b].x, nodes[b].y);
            hctx.strokeStyle = 'rgba(0,240,255,' + alpha + ')';
            hctx.lineWidth = 0.6;
            hctx.stroke();
          }
        }
      }

      requestAnimationFrame(heroFrame);
    }
    requestAnimationFrame(heroFrame);

    /* ── NETWORK CANVAS — ORBITAL MAP ── */
    var nc = document.getElementById('networkCanvas');
    var nctx = nc.getContext('2d');
    var nW, nH;
    var nData = [
      { label: 'JP', angle: 0, orbit: 0, color: '#00F0FF', primary: true },
      { label: 'SG', angle: 0.9, orbit: 1, color: '#00F0FF' },
      { label: 'AU', angle: 2.1, orbit: 1, color: '#00F0FF' },
      { label: 'KR', angle: 3.5, orbit: 0.7, color: '#00F0FF' },
      { label: 'HK', angle: 4.5, orbit: 1, color: '#00F0FF' },
      { label: 'IN', angle: 5.5, orbit: 0.85, color: '#00F0FF' }
    ];

    function nResize() {
      var container = nc.parentElement;
      nW = nc.width = container.offsetWidth;
      nH = nc.height = container.offsetHeight;
    }
    nResize();
    window.addEventListener('resize', nResize);

    function drawNetwork(t) {
      nctx.clearRect(0, 0, nW, nH);
      var cx = nW / 2, cy = nH / 2;
      var maxR = Math.min(nW, nH) * 0.38;
      var time = t * 0.001;

      // orbit rings
      for (var ring = 1; ring <= 3; ring++) {
        var rr = maxR * (ring / 3);
        var pulse = Math.sin(time * 0.5 + ring) * 0.05 + 1;
        nctx.beginPath();
        nctx.arc(cx, cy, rr * pulse, 0, Math.PI * 2);
        nctx.strokeStyle = 'rgba(0,240,255,' + (0.06 + (3 - ring) * 0.03) + ')';
        nctx.lineWidth = 1;
        nctx.stroke();
      }

      // calc positions
      var positions = [];
      for (var i = 0; i < nData.length; i++) {
        var d = nData[i];
        if (d.primary) {
          positions.push({ x: cx, y: cy, label: d.label, color: d.color, primary: true });
        } else {
          var a = d.angle + time * 0.08;
          var r = maxR * d.orbit;
          positions.push({
            x: cx + Math.cos(a) * r,
            y: cy + Math.sin(a) * r,
            label: d.label,
            color: d.color,
            primary: false
          });
        }
      }

      // connection lines
      for (var j = 1; j < positions.length; j++) {
        var p0 = positions[0], p1 = positions[j];
        var grad = nctx.createLinearGradient(p0.x, p0.y, p1.x, p1.y);
        grad.addColorStop(0, 'rgba(0,240,255,0.25)');
        grad.addColorStop(1, 'rgba(0,240,255,0.03)');
        nctx.beginPath();
        nctx.moveTo(p0.x, p0.y);
        nctx.lineTo(p1.x, p1.y);
        nctx.strokeStyle = grad;
        nctx.lineWidth = 1;
        nctx.setLineDash([4, 6]);
        nctx.stroke();
        nctx.setLineDash([]);

        // traveling pulse
        var prog = ((time * 0.3 + j * 0.5) % 1);
        var px = p0.x + (p1.x - p0.x) * prog;
        var py = p0.y + (p1.y - p0.y) * prog;
        nctx.beginPath();
        nctx.arc(px, py, 2, 0, Math.PI * 2);
        nctx.fillStyle = 'rgba(0,240,255,0.7)';
        nctx.fill();
      }

      // draw nodes
      for (var k = 0; k < positions.length; k++) {
        var nd = positions[k];
        var sz = nd.primary ? 32 : 20;

        // outer glow
        var glowR = sz * 2;
        var glow = nctx.createRadialGradient(nd.x, nd.y, 0, nd.x, nd.y, glowR);
        glow.addColorStop(0, nd.primary ? 'rgba(0,240,255,0.15)' : 'rgba(0,240,255,0.08)');
        glow.addColorStop(1, 'rgba(0,240,255,0)');
        nctx.beginPath();
        nctx.arc(nd.x, nd.y, glowR, 0, Math.PI * 2);
        nctx.fillStyle = glow;
        nctx.fill();

        // node circle
        nctx.beginPath();
        nctx.arc(nd.x, nd.y, sz / 2, 0, Math.PI * 2);
        nctx.fillStyle = nd.primary ? '#00F0FF' : 'rgba(6,8,12,0.9)';
        nctx.fill();
        nctx.strokeStyle = 'rgba(0,240,255,' + (nd.primary ? '0.8' : '0.4') + ')';
        nctx.lineWidth = nd.primary ? 2 : 1;
        nctx.stroke();

        // label
        nctx.font = (nd.primary ? '700 13px' : '500 10px') + ' "Space Mono", monospace';
        nctx.fillStyle = nd.primary ? '#06080C' : '#00F0FF';
        nctx.textAlign = 'center';
        nctx.textBaseline = 'middle';
        nctx.fillText(nd.label, nd.x, nd.y + (nd.primary ? 0.5 : 0));
      }

      requestAnimationFrame(drawNetwork);
    }
    requestAnimationFrame(drawNetwork);
  })();
  </script>
